**Goal:** Enable correct stereoscopic rendering of Gaussian Splats on Apple Vision Pro via Unity PolySpatial Metal mode with Single-Pass Instanced rendering.

**Architecture:** Replace `DrawProcedural` with `Graphics.RenderMeshPrimitives` using a pre-built batched mesh (following gsplat-unity pattern). Move 2D covariance calculation from the CalcViewData compute prepass into the vertex shader so Unity's stereo rendering pipeline automatically provides per-eye view/projection matrices. **Tech Stack:** Unity 6 URP, PolySpatial visionOS (Metal mode), Single-Pass Instanced stereo, ComputeShader (sort only), HLSL

**Reference:** [gsplat-unity](https://github.com/wuyize25/gsplat-unity) for RenderMeshPrimitives + stereo pattern

## Task 1: Create VR Splat Mesh Generator

Creates a batched mesh where each "instance" contains N splat quads. Vertex positions encode local splat index for buffer lookup. This replaces `DrawProcedural`.

**Files:**
- Create: `Assets/4DGS/Runtime/GaussianSplatVRMesh.cs`
- Create: `Assets/4DGS/Runtime/GaussianSplatVRMesh.cs.meta` (auto-generated by Unity)

**Step 1: Write GaussianSplatVRMesh.cs**

```csharp
using System;
using UnityEngine;

namespace GaussianSplatting.Runtime
{
    /// <summary>
    /// Generates a batched mesh for VR stereo-compatible splat rendering.
    /// Each mesh instance contains <see cref="SplatsPerBatch"/> quads.
    /// Vertex position.z encodes the local splat index within the batch.
    /// </summary>
    public static class GaussianSplatVRMesh
    {
        public const int SplatsPerBatch = 128;

        private static Mesh _cachedMesh;

        public static Mesh GetOrCreate()
        {
            if (_cachedMesh != null)
                return _cachedMesh;

            _cachedMesh = CreateMesh(SplatsPerBatch);
            return _cachedMesh;
        }

        private static Mesh CreateMesh(int splatsPerBatch)
        {
            var positions = new Vector3[splatsPerBatch * 4];
            var indices = new int[splatsPerBatch * 6];

            for (int i = 0; i < splatsPerBatch; i++)
            {
                // Encode local splat index in z component (reinterpreted as uint in shader)
                uint ui = (uint)i;
                float z;
                unsafe { z = *(float*)&ui; }

                int v = i * 4;
                positions[v + 0] = new Vector3(-1, -1, z);
                positions[v + 1] = new Vector3( 1, -1, z);
                positions[v + 2] = new Vector3(-1,  1, z);
                positions[v + 3] = new Vector3( 1,  1, z);

                int t = i * 6;
                indices[t + 0] = v + 0;
                indices[t + 1] = v + 1;
                indices[t + 2] = v + 2;
                indices[t + 3] = v + 1;
                indices[t + 4] = v + 3;
                indices[t + 5] = v + 2;
            }

            var mesh = new Mesh
            {
                name = "GaussianSplatVRBatch",
                vertices = positions,
                triangles = indices,
                bounds = new Bounds(Vector3.zero, Vector3.one * 10000f),
                hideFlags = HideFlags.HideAndDontSave
            };
            return mesh;
        }

        public static void Dispose()
        {
            if (_cachedMesh != null)
            {
                UnityEngine.Object.DestroyImmediate(_cachedMesh);
                _cachedMesh = null;
            }
        }

        public static int CalcInstanceCount(int splatCount)
        {
            return Mathf.CeilToInt(splatCount / (float)SplatsPerBatch);
        }
    }
}
```

**Step 1: Write RenderGaussianSplatsVR.shader**

```hlsl
// SPDX-License-Identifier: MIT
Shader "Gaussian Splatting/Render Splats VR"
{
    SubShader
    {
        Tags { "RenderType"="Transparent" "Queue"="Transparent" }

        Pass
        {
            ZWrite Off
            Blend One OneMinusSrcAlpha
            Cull Off

            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma require compute
            #pragma use_dxc
            #pragma multi_compile_instancing

            #include "UnityCG.cginc"
            #include "GaussianSplatDecode.hlsl"
            #include "GaussianSplatting.hlsl"

            int _SplatCount;
            int _SplatInstanceSize;
            int _SplatFormat;
            int _SplatChunkCount;
            int _SHOrder;
            int _SHOnly;
            float _SplatScale;
            float _SplatOpacityScale;
            float4x4 _MatrixObjectToWorld;
            float4x4 _MatrixWorldToObject;

            StructuredBuffer<uint> _OrderBuffer;
            ByteAddressBuffer _SplatPos;
            ByteAddressBuffer _SplatOther;
            ByteAddressBuffer _SplatSH;
            Texture2D<half4> _SplatColor;
            StructuredBuffer<SplatChunkData> _SplatChunks;

            // Animation data (optional, set by GaussianAnimator)
            int _AnimDataValid;
            StructuredBuffer<float4> _AnimOutputData;

            struct appdata
            {
                float4 vertex : POSITION;
                UNITY_VERTEX_INPUT_INSTANCE_ID
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                half4 col : COLOR0;
                float2 centerOffset : TEXCOORD0;
                UNITY_VERTEX_OUTPUT_STEREO
            };

            v2f vert(appdata v)
            {
                v2f o;
                UNITY_SETUP_INSTANCE_ID(v);
                UNITY_INITIALIZE_OUTPUT(v2f, o);
                UNITY_INITIALIZE_VERTEX_OUTPUT_STEREO(o);

                // Decode splat index from mesh data
                // v.vertex.xy = quad corner (-1,-1) to (1,1)
                // v.vertex.z = local splat index (reinterpreted as uint)
                uint localIdx = asuint(v.vertex.z);

                #if defined(UNITY_INSTANCING_ENABLED) || defined(UNITY_PROCEDURAL_INSTANCING_ENABLED) || defined(UNITY_STEREO_INSTANCING_ENABLED)
                uint globalIdx = unity_InstanceID * (uint)_SplatInstanceSize + localIdx;
                #else
                uint globalIdx = v.instanceID * (uint)_SplatInstanceSize + localIdx;
                #endif

                // Bounds check
                if (globalIdx >= (uint)_SplatCount)
                {
                    o.vertex = asfloat(0x7fc00000); // NaN discard
                    return o;
                }

                // Get sorted splat index
                uint splatIdx = _OrderBuffer[globalIdx];

                // Decode splat data from compressed buffers
                DecodedSplat splat = SplatDecodeFromBuffers(
                    _SplatPos, _SplatOther, _SplatSH, _SplatColor,
                    _SplatChunks, (uint)_SplatChunkCount,
                    splatIdx, (uint)_SplatFormat);

                // Apply animation
                float animOpacityMul = 1.0;
                float animScaleMul = 1.0;
                float3 animColorTint = float3(1, 1, 1);
                float animColorBlend = 0;

                if (_AnimDataValid != 0)
                {
                    uint animBase = splatIdx * 3;
                    float4 animPosOp = _AnimOutputData[animBase + 0];
                    float4 animScaleColor = _AnimOutputData[animBase + 1];
                    float4 animTint = _AnimOutputData[animBase + 2];
                    splat.pos += animPosOp.xyz;
                    animOpacityMul = animPosOp.w;
                    animScaleMul = animScaleColor.x;
                    animColorBlend = animScaleColor.y;
                    animColorTint = animTint.xyz;
                }

                // Transform to world and then to view/clip
                // _MatrixObjectToWorld is the renderer's localToWorldMatrix
                float3 worldPos = mul(_MatrixObjectToWorld, float4(splat.pos, 1)).xyz;

                // UNITY_MATRIX_V is automatically per-eye in stereo mode
                float4x4 viewMat = UNITY_MATRIX_V;
                float4x4 projMat = UNITY_MATRIX_P;
                float3 viewPos = mul(viewMat, float4(worldPos, 1)).xyz;

                // Behind camera check
                if (viewPos.z > 0)
                {
                    o.vertex = asfloat(0x7fc00000);
                    return o;
                }

                float4 clipPos = mul(projMat, float4(viewPos, 1));
                clipPos.z = clamp(clipPos.z, -abs(clipPos.w), abs(clipPos.w));

                // Opacity check
                float finalOpacity = splat.opacity * _SplatOpacityScale * animOpacityMul;
                if (finalOpacity < 1.0 / 255.0)
                {
                    o.vertex = asfloat(0x7fc00000);
                    return o;
                }

                // Compute 3D covariance
                float3 boxSize = splat.scale * animScaleMul;
                float3x3 rotScaleMat = CalcMatrixFromRotationScale(splat.rot, boxSize);
                float3 cov3d0, cov3d1;
                CalcCovariance3D(rotScaleMat, cov3d0, cov3d1);
                float splatScale2 = _SplatScale * _SplatScale;
                cov3d0 *= splatScale2;
                cov3d1 *= splatScale2;

                // Project to 2D covariance
                // Use the per-eye view matrix for correct stereo projection
                float4x4 matMV = mul(viewMat, _MatrixObjectToWorld);
                float3 cov2d = CalcCovariance2D(
                    splat.pos, cov3d0, cov3d1,
                    matMV, projMat, _ScreenParams);

                // Decompose to ellipse axes
                float2 axis1, axis2;
                DecomposeCovariance(cov2d, axis1, axis2);

                // Sub-pixel culling
                float area = length(axis1) * length(axis2);
                if (area < 1.0)
                {
                    o.vertex = asfloat(0x7fc00000);
                    return o;
                }

                // SH color evaluation (per-eye view direction)
                float3 camWorldPos = _WorldSpaceCameraPos.xyz;
                float3 worldViewDir = camWorldPos - worldPos;
                float3 objViewDir = normalize(mul((float3x3)_MatrixWorldToObject, worldViewDir));

                SplatSHData shData;
                [unroll]
                for (int i = 0; i < 15; i++)
                    shData.col[i] = splat.sh[i];

                half3 baseColor = splat.color;
                half3 shColor = ShadeSH(shData, (half3)objViewDir, _SHOrder, _SHOnly != 0);
                half3 finalColor = baseColor + shColor;

                // Apply animation color tint
                if (animColorBlend > 0)
                    finalColor = lerp(finalColor, (half3)animColorTint, (half)animColorBlend);

                finalColor = max(finalColor, half3(0, 0, 0));

                o.col = half4(finalColor, (half)finalOpacity);

                // Expand quad using ellipse axes
                float2 quadPos = v.vertex.xy; // (-1,-1) to (1,1)
                o.centerOffset = quadPos;

                float2 deltaScreen = (quadPos.x * axis1 + quadPos.y * axis2) * 2 / _ScreenParams.xy;
                o.vertex = clipPos;
                o.vertex.xy += deltaScreen * clipPos.w;

                return o;
            }

            half4 frag(v2f i) : SV_Target
            {
                UNITY_SETUP_STEREO_EYE_INDEX_POST_VERTEX(i);

                float power = -dot(i.centerOffset, i.centerOffset);
                half alpha = exp(power);
                alpha = saturate(alpha * i.col.a);

                if (alpha < 1.0 / 255.0)
                    discard;

                return half4(i.col.rgb * alpha, alpha);
            }
            ENDHLSL
        }
    }
}
```

```csharp
private void RenderSplatsVR(Camera cam, GaussianSplatRenderer gs, MaterialPropertyBlock mpb)
{
    if (_vrMaterial == null)
        return;

    int splatCount = gs.EffectiveSplatCount;
    if (splatCount <= 0)
        return;

    _vrMpb.Clear();

    // Set splat data buffers
    gs.SetAssetDataOnMaterial(_vrMpb);

    // Set sort output
    _vrMpb.SetBuffer(GaussianSplatRenderer.Props.OrderBuffer, gs.GpuSortKeys);

    // Set uniforms
    _vrMpb.SetInteger(GaussianSplatRenderer.Props.SplatCount, splatCount);
    _vrMpb.SetInteger("_SplatInstanceSize", GaussianSplatVRMesh.SplatsPerBatch);
    _vrMpb.SetFloat(GaussianSplatRenderer.Props.SplatScale, gs.splatScale);
    _vrMpb.SetFloat(GaussianSplatRenderer.Props.SplatOpacityScale, gs.opacityScale);
    _vrMpb.SetInteger(GaussianSplatRenderer.Props.SHOrder, gs.shOrder);
    _vrMpb.SetInteger(GaussianSplatRenderer.Props.SHOnly, gs.shOnly ? 1 : 0);
    _vrMpb.SetMatrix(GaussianSplatRenderer.Props.MatrixObjectToWorld, gs.transform.localToWorldMatrix);
    _vrMpb.SetMatrix(GaussianSplatRenderer.Props.MatrixWorldToObject, gs.transform.worldToLocalMatrix);

    // Animation data
    bool hasAnim = gs.AnimOutputBuffer != null;
    _vrMpb.SetInteger(GaussianSplatRenderer.Props.AnimDataValid, hasAnim ? 1 : 0);
    if (hasAnim)
        _vrMpb.SetBuffer(GaussianSplatRenderer.Props.AnimOutputData, gs.AnimOutputBuffer);

    // Render
    var mesh = GaussianSplatVRMesh.GetOrCreate();
    int instanceCount = GaussianSplatVRMesh.CalcInstanceCount(splatCount);

    var rp = new RenderParams(_vrMaterial)
    {
        worldBounds = gs.GetWorldBounds(),
        matProps = _vrMpb,
        layer = gs.gameObject.layer,
        renderingLayerMask = GraphicsSettings.defaultRenderingLayerMask,
    };

    Graphics.RenderMeshPrimitives(rp, mesh, 0, instanceCount);
}
```
